50 /// A pass to convert gpu.launch_func operations into a sequence of GPU
51 /// runtime calls. Currently it supports CUDA and ROCm (HIP).
52 ///
53 /// In essence, a gpu.launch_func operations gets compiled into the following
54 /// sequence of runtime calls:
55 ///
56 /// * moduleLoad -- loads the module given the cubin / hsaco data
57 /// * moduleGetFunction -- gets a handle to the actual kernel function
58 /// * getStreamHelper -- initializes a new compute stream on GPU
59 /// * launchKernel -- launches the kernel on a stream
60 /// * streamSynchronize -- waits for operations on the stream to finish
61 ///
62 /// Intermediate data structures are allocated on the stack.
63 class GpuLaunchFuncToGpuRuntimeCallsPass
64  : public ConvertGpuLaunchFuncToGpuRuntimeCallsBase<
65  GpuLaunchFuncToGpuRuntimeCallsPass> {

71  }

75  llvmVoidType = LLVM::LLVMType::getVoidTy(llvmDialect);
76  llvmPointerType = LLVM::LLVMType::getInt8PtrTy(llvmDialect);
77  llvmPointerPointerType = llvmPointerType.getPointerTo();
78  llvmInt8Type = LLVM::LLVMType::getInt8Ty(llvmDialect);

161 // Adds declarations for the needed helper functions from the runtime wrappers.
162 // The types in comments give the actual types expected/returned but the API
163 // uses void pointers. This is fine as they have the same linkage in C.
164 void GpuLaunchFuncToGpuRuntimeCallsPass::declareGpuRuntimeFunctions(
165  Location loc) {
166  ModuleOp module = getOperation();
167  OpBuilder builder(module.getBody()->getTerminator());
168  if (!module.lookupSymbol(kGpuModuleLoadName)) {
169  builder.create<LLVM::LLVMFuncOp>(
170  loc, kGpuModuleLoadName,
171  LLVM::LLVMType::getFunctionTy(
172  getGpuRuntimeResultType(),
173  {
174  getPointerPointerType(), /* CUmodule *module */
175  getPointerType() /* void *cubin */
176  },
177  /*isVarArg=*/false));
178  }
179  if (!module.lookupSymbol(kGpuModuleGetFunctionName)) {
180  // The helper uses void* instead of CUDA's opaque CUmodule and
181  // CUfunction, or ROCm (HIP)'s opaque hipModule_t and hipFunction_t.
182  builder.create<LLVM::LLVMFuncOp>(
183  loc, kGpuModuleGetFunctionName,
184  LLVM::LLVMType::getFunctionTy(
185  getGpuRuntimeResultType(),
186  {
187  getPointerPointerType(), /* void **function */
188  getPointerType(), /* void *module */
189  getPointerType() /* char *name */
190  },
191  /*isVarArg=*/false));
192  }
193  if (!module.lookupSymbol(kGpuLaunchKernelName)) {
194  // Other than the CUDA or ROCm (HIP) api, the wrappers use uintptr_t to
195  // match the LLVM type if MLIR's index type, which the GPU dialect uses.
196  // Furthermore, they use void* instead of CUDA's opaque CUfunction and
197  // CUstream, or ROCm (HIP)'s opaque hipFunction_t and hipStream_t.
198  builder.create<LLVM::LLVMFuncOp>(
199  loc, kGpuLaunchKernelName,
200  LLVM::LLVMType::getFunctionTy(
201  getGpuRuntimeResultType(),
202  {
203  getPointerType(), /* void* f */
204  getIntPtrType(), /* intptr_t gridXDim */
205  getIntPtrType(), /* intptr_t gridyDim */
206  getIntPtrType(), /* intptr_t gridZDim */
207  getIntPtrType(), /* intptr_t blockXDim */
208  getIntPtrType(), /* intptr_t blockYDim */
209  getIntPtrType(), /* intptr_t blockZDim */
210  getInt32Type(), /* unsigned int sharedMemBytes */
211  getPointerType(), /* void *hstream */
212  getPointerPointerType(), /* void **kernelParams */
213  getPointerPointerType() /* void **extra */
214  },
215  /*isVarArg=*/false));
216  }
217  if (!module.lookupSymbol(kGpuGetStreamHelperName)) {
218  // Helper function to get the current GPU compute stream. Uses void*
219  // instead of CUDA's opaque CUstream, or ROCm (HIP)'s opaque hipStream_t.
220  builder.create<LLVM::LLVMFuncOp>(
221  loc, kGpuGetStreamHelperName,
222  LLVM::LLVMType::getFunctionTy(getPointerType(), /*isVarArg=*/false));
223  }
224  if (!module.lookupSymbol(kGpuStreamSynchronizeName)) {
225  builder.create<LLVM::LLVMFuncOp>(
226  loc, kGpuStreamSynchronizeName,
227  LLVM::LLVMType::getFunctionTy(getGpuRuntimeResultType(),
228  getPointerType() /* CUstream stream */,
229  /*isVarArg=*/false));
230  }
231  if (!module.lookupSymbol(kGpuMemHostRegisterName)) {
232  builder.create<LLVM::LLVMFuncOp>(
233  loc, kGpuMemHostRegisterName,
234  LLVM::LLVMType::getFunctionTy(getVoidType(),
235  {
236  getPointerType(), /* void *ptr */
237  getInt64Type() /* int64 sizeBytes*/
238  },
239  /*isVarArg=*/false));
240  }
241 }

243 /// Emits the IR with the following structure:
244 ///
245 /// %data = llvm.alloca 1 x type-of(<param>)
246 /// llvm.store <param>, %data
247 /// %typeErased = llvm.bitcast %data to !llvm<"i8*">
248 /// %addr = llvm.getelementptr <list>[<pos>]
249 /// llvm.store %typeErased, %addr
250 ///
251 /// This is necessary to construct the list of arguments passed to the kernel
252 /// function as accepted by cuLaunchKernel, i.e. as a void** that points to list
253 /// of stack-allocated type-erased pointers to the actual arguments.
254 void GpuLaunchFuncToGpuRuntimeCallsPass::addParamToList(OpBuilder &builder,
255  Location loc,
256  Value param, Value list,
257  unsigned pos,
258  Value one) {
259  auto memLocation = builder.create<LLVM::AllocaOp>(
260  loc, param.getType().cast<LLVM::LLVMType>().getPointerTo(), one,
261  /*alignment=*/1);
262  builder.create<LLVM::StoreOp>(loc, param, memLocation);
263  auto casted =
264  builder.create<LLVM::BitcastOp>(loc, getPointerType(), memLocation);
265 
266  auto index = builder.create<LLVM::ConstantOp>(loc, getInt32Type(),
267  builder.getI32IntegerAttr(pos));
268  auto gep = builder.create<LLVM::GEPOp>(loc, getPointerPointerType(), list,
269  ArrayRef<Value>{index});
270  builder.create<LLVM::StoreOp>(loc, casted, gep);
271 }

273 // Generates a parameters array to be used with a CUDA / ROCm (HIP) kernel
274 // launch call. The arguments are extracted from the launchOp.
275 // The generated code is essentially as follows:
276 //
277 // %array = alloca(numparams * sizeof(void *))
278 // for (i : [0, NumKernelOperands))
279 // %array[i] = cast<void*>(KernelOperand[i])
280 // return %array
281 Value GpuLaunchFuncToGpuRuntimeCallsPass::setupParamsArray(
282  gpu::LaunchFuncOp launchOp, OpBuilder &builder) {
283 
284  // Get the launch target.
285  auto gpuFunc = SymbolTable::lookupNearestSymbolFrom<LLVM::LLVMFuncOp>(
286  launchOp, launchOp.kernel());
287  if (!gpuFunc)
288  return {};
289 
290  unsigned numArgs = gpuFunc.getNumArguments();
291 
292  auto numKernelOperands = launchOp.getNumKernelOperands();
293  Location loc = launchOp.getLoc();
294  auto one = builder.create<LLVM::ConstantOp>(loc, getInt32Type(),
295  builder.getI32IntegerAttr(1));
296  auto arraySize = builder.create<LLVM::ConstantOp>(
297  loc, getInt32Type(), builder.getI32IntegerAttr(numArgs));
298  auto array = builder.create<LLVM::AllocaOp>(loc, getPointerPointerType(),
299  arraySize, /*alignment=*/0);
300 
301  unsigned pos = 0;
302  for (unsigned idx = 0; idx < numKernelOperands; ++idx) {
303  auto operand = launchOp.getKernelOperand(idx);
304  auto llvmType = operand.getType().cast<LLVM::LLVMType>();
305 
306  // Assume all struct arguments come from MemRef. If this assumption does not
307  // hold anymore then we `launchOp` to lower from MemRefType and not after
308  // LLVMConversion has taken place and the MemRef information is lost.
309  if (!llvmType.isStructTy()) {
310  addParamToList(builder, loc, operand, array, pos++, one);
311  continue;
312  }
313 
314  // Put individual components of a memref descriptor into the flat argument
315  // list. We cannot use unpackMemref from LLVM lowering here because we have
316  // no access to MemRefType that had been lowered away.
317  for (int32_t j = 0, ej = llvmType.getStructNumElements(); j < ej; ++j) {
318  auto elemType = llvmType.getStructElementType(j);
319  if (elemType.isArrayTy()) {
320  for (int32_t k = 0, ek = elemType.getArrayNumElements(); k < ek; ++k) {
321  Value elem = builder.create<LLVM::ExtractValueOp>(
322  loc, elemType.getArrayElementType(), operand,
323  builder.getI32ArrayAttr({j, k}));
324  addParamToList(builder, loc, elem, array, pos++, one);
325  }
326  } else {
327  assert((elemType.isIntegerTy() || elemType.isFloatTy() ||
328  elemType.isDoubleTy() || elemType.isPointerTy()) &&
329  "expected scalar type");
330  Value strct = builder.create<LLVM::ExtractValueOp>(
331  loc, elemType, operand, builder.getI32ArrayAttr(j));
332  addParamToList(builder, loc, strct, array, pos++, one);
333  }
334  }
335  }
336 
337  return array;
338 }

340 // Generates an LLVM IR dialect global that contains the name of the given
341 // kernel function as a C string, and returns a pointer to its beginning.
342 // The code is essentially:
343 //
344 // llvm.global constant @kernel_name("function_name\00")
345 // func(...) {
346 // %0 = llvm.addressof @kernel_name
347 // %1 = llvm.constant (0 : index)
348 // %2 = llvm.getelementptr %0[%1, %1] : !llvm<"i8*">
349 // }
350 Value GpuLaunchFuncToGpuRuntimeCallsPass::generateKernelNameConstant(
351  StringRef moduleName, StringRef name, Location loc, OpBuilder &builder) {
352  // Make sure the trailing zero is included in the constant.
353  std::vector<char> kernelName(name.begin(), name.end());
354  kernelName.push_back('\0');
355 
356  std::string globalName =
357  std::string(llvm::formatv("{0}_{1}_kernel_name", moduleName, name));
358  return LLVM::createGlobalString(
359  loc, builder, globalName, StringRef(kernelName.data(), kernelName.size()),
360  LLVM::Linkage::Internal, llvmDialect);
361 }

363 // Emits LLVM IR to launch a kernel function. Expects the module that contains
364 // the compiled kernel function as a cubin in the 'nvvm.cubin' attribute, or a
365 // hsaco in the 'rocdl.hsaco' attribute of the kernel function in the IR.
366 //
367 // %0 = call %binarygetter
368 // %1 = alloca sizeof(void*)
369 // call %moduleLoad(%2, %1)
370 // %2 = alloca sizeof(void*)
371 // %3 = load %1
372 // %4 = <see generateKernelNameConstant>
373 // call %moduleGetFunction(%2, %3, %4)
374 // %5 = call %getStreamHelper()
375 // %6 = load %2
376 // %7 = <see setupParamsArray>
377 // call %launchKernel(%6, <launchOp operands 0..5>, 0, %5, %7, nullptr)
378 // call %streamSynchronize(%5)
379 void GpuLaunchFuncToGpuRuntimeCallsPass::translateGpuLaunchCalls(
380  mlir::gpu::LaunchFuncOp launchOp) {
381  OpBuilder builder(launchOp);
382  Location loc = launchOp.getLoc();
383  declareGpuRuntimeFunctions(loc);
384 
385  auto zero = builder.create<LLVM::ConstantOp>(loc, getInt32Type(),
386  builder.getI32IntegerAttr(0));
387  // Create an LLVM global with CUBIN extracted from the kernel annotation and
388  // obtain a pointer to the first byte in it.
389  auto kernelModule = getOperation().lookupSymbol<gpu::GPUModuleOp>(
390  launchOp.getKernelModuleName());
391  assert(kernelModule && "expected a kernel module");
392 
393  auto binaryAttr = kernelModule.getAttrOfType<StringAttr>(gpuBinaryAnnotation);
394  if (!binaryAttr) {
395  kernelModule.emitOpError()
396  << "missing " << gpuBinaryAnnotation << " attribute";
397  return signalPassFailure();
398  }
399 
400  SmallString<128> nameBuffer(kernelModule.getName());
401  nameBuffer.append(kGpuBinaryStorageSuffix);
402  Value data = LLVM::createGlobalString(
403  loc, builder, nameBuffer.str(), binaryAttr.getValue(),
404  LLVM::Linkage::Internal, getLLVMDialect());
405 
406  // Emit the load module call to load the module data. Error checking is done
407  // in the called helper function.
408  auto gpuModule = allocatePointer(builder, loc);
409  auto gpuModuleLoad =
410  getOperation().lookupSymbol<LLVM::LLVMFuncOp>(kGpuModuleLoadName);
411  builder.create<LLVM::CallOp>(loc, ArrayRef<Type>{getGpuRuntimeResultType()},
412  builder.getSymbolRefAttr(gpuModuleLoad),
413  ArrayRef<Value>{gpuModule, data});
414  // Get the function from the module. The name corresponds to the name of
415  // the kernel function.
416  auto gpuOwningModuleRef =
417  builder.create<LLVM::LoadOp>(loc, getPointerType(), gpuModule);
418  auto kernelName = generateKernelNameConstant(
419  launchOp.getKernelModuleName(), launchOp.getKernelName(), loc, builder);
420  auto gpuFunction = allocatePointer(builder, loc);
421  auto gpuModuleGetFunction =
422  getOperation().lookupSymbol<LLVM::LLVMFuncOp>(kGpuModuleGetFunctionName);
423  builder.create<LLVM::CallOp>(
424  loc, ArrayRef<Type>{getGpuRuntimeResultType()},
425  builder.getSymbolRefAttr(gpuModuleGetFunction),
426  ArrayRef<Value>{gpuFunction, gpuOwningModuleRef, kernelName});
427  // Grab the global stream needed for execution.
428  auto gpuGetStreamHelper =
429  getOperation().lookupSymbol<LLVM::LLVMFuncOp>(kGpuGetStreamHelperName);
430  auto gpuStream = builder.create<LLVM::CallOp>(
431  loc, ArrayRef<Type>{getPointerType()},
432  builder.getSymbolRefAttr(gpuGetStreamHelper), ArrayRef<Value>{});
433  // Invoke the function with required arguments.
434  auto gpuLaunchKernel =
435  getOperation().lookupSymbol<LLVM::LLVMFuncOp>(kGpuLaunchKernelName);
436  auto gpuFunctionRef =
437  builder.create<LLVM::LoadOp>(loc, getPointerType(), gpuFunction);
438  auto paramsArray = setupParamsArray(launchOp, builder);
439  if (!paramsArray) {
440  launchOp.emitOpError() << "cannot pass given parameters to the kernel";
441  return signalPassFailure();
442  }
443  auto nullpointer =
444  builder.create<LLVM::IntToPtrOp>(loc, getPointerPointerType(), zero);
445  builder.create<LLVM::CallOp>(
446  loc, ArrayRef<Type>{getGpuRuntimeResultType()},
447  builder.getSymbolRefAttr(gpuLaunchKernel),
448  ArrayRef<Value>{gpuFunctionRef, launchOp.getOperand(0),
449  launchOp.getOperand(1), launchOp.getOperand(2),
450  launchOp.getOperand(3), launchOp.getOperand(4),
451  launchOp.getOperand(5), zero, /* sharedMemBytes */
452  gpuStream.getResult(0), /* stream */
453  paramsArray, /* kernel params */
454  nullpointer /* extra */});
455  // Sync on the stream to make it synchronous.
456  auto gpuStreamSync =
457  getOperation().lookupSymbol<LLVM::LLVMFuncOp>(kGpuStreamSynchronizeName);
458  builder.create<LLVM::CallOp>(loc, ArrayRef<Type>{getGpuRuntimeResultType()},
459  builder.getSymbolRefAttr(gpuStreamSync),
460  ArrayRef<Value>(gpuStream.getResult(0)));
461  launchOp.erase();
462 }
