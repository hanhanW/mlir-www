34 ArrayRef<Extension> spirv::getImpliedExtensions(Version version) {
35  // Note: the following lists are from "Appendix A: Changes" of the spec.
36 
37 #define V_1_3_IMPLIED_EXTS \
38  Extension::SPV_KHR_shader_draw_parameters, Extension::SPV_KHR_16bit_storage, \
39  Extension::SPV_KHR_device_group, Extension::SPV_KHR_multiview, \
40  Extension::SPV_KHR_storage_buffer_storage_class, \
41  Extension::SPV_KHR_variable_pointers
42 
43 #define V_1_4_IMPLIED_EXTS \
44  Extension::SPV_KHR_no_integer_wrap_decoration, \
45  Extension::SPV_GOOGLE_decorate_string, \
46  Extension::SPV_GOOGLE_hlsl_functionality1, \
47  Extension::SPV_KHR_float_controls
48 
49 #define V_1_5_IMPLIED_EXTS \
50  Extension::SPV_KHR_8bit_storage, Extension::SPV_EXT_descriptor_indexing, \
51  Extension::SPV_EXT_shader_viewport_index_layer, \
52  Extension::SPV_EXT_physical_storage_buffer, \
53  Extension::SPV_KHR_physical_storage_buffer, \
54  Extension::SPV_KHR_vulkan_memory_model
55 
56  switch (version) {
57  default:
58  return {};
59  case Version::V_1_3: {
60  // The following manual ArrayRef constructor call is to satisfy GCC 5.
61  static const Extension exts[] = {V_1_3_IMPLIED_EXTS};
62  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
63  }
64  case Version::V_1_4: {
65  static const Extension exts[] = {V_1_3_IMPLIED_EXTS, V_1_4_IMPLIED_EXTS};
66  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
67  }
68  case Version::V_1_5: {
69  static const Extension exts[] = {V_1_3_IMPLIED_EXTS, V_1_4_IMPLIED_EXTS,
70  V_1_5_IMPLIED_EXTS};
71  return ArrayRef<Extension>(exts, llvm::array_lengthof(exts));
72  }
73  }
74 
75 #undef V_1_5_IMPLIED_EXTS
76 #undef V_1_4_IMPLIED_EXTS
77 #undef V_1_3_IMPLIED_EXTS
78 }

735 void ScalarType::getCapabilities(

737  Optional<StorageClass> storage) {

739 
740  // 8- or 16-bit integer/floating-point numbers will require extra capabilities
741  // to appear in interface storage classes. See SPV_KHR_16bit_storage and
742  // SPV_KHR_8bit_storage for more details.
743 
744 #define STORAGE_CASE(storage, cap8, cap16) \
745  case StorageClass::storage: { \
746  if (bitwidth == 8) { \
747  static const Capability caps[] = {Capability::cap8}; \
748  ArrayRef<Capability> ref(caps, llvm::array_lengthof(caps)); \
749  capabilities.push_back(ref); \
750  } else if (bitwidth == 16) { \
751  static const Capability caps[] = {Capability::cap16}; \
752  ArrayRef<Capability> ref(caps, llvm::array_lengthof(caps)); \
753  capabilities.push_back(ref); \
754  } \
755  } break
756 
757  if (storage) {
758  switch (*storage) {
759  STORAGE_CASE(PushConstant, StoragePushConstant8, StoragePushConstant16);
760  STORAGE_CASE(StorageBuffer, StorageBuffer8BitAccess,
761  StorageBuffer16BitAccess);
762  STORAGE_CASE(Uniform, UniformAndStorageBuffer8BitAccess,
763  StorageUniform16);
764  case StorageClass::Input:
765  case StorageClass::Output:
766  if (bitwidth == 16) {
767  static const Capability caps[] = {Capability::StorageInputOutput16};
768  ArrayRef<Capability> ref(caps, llvm::array_lengthof(caps));
769  capabilities.push_back(ref);
770  }
771  break;
772  default:
773  break;
774  }
775  return;
776  }

778 
779  // For other non-interface storage classes, require a different set of
780  // capabilities for special bitwidths.
781 
782 #define WIDTH_CASE(type, width) \
783  case width: { \
784  static const Capability caps[] = {Capability::type##width}; \
785  ArrayRef<Capability> ref(caps, llvm::array_lengthof(caps)); \
786  capabilities.push_back(ref); \
787  } break
788 
789  if (auto intType = dyn_cast<IntegerType>()) {
790  switch (bitwidth) {
791  case 32:
792  case 1:
793  break;
794  WIDTH_CASE(Int, 8);
795  WIDTH_CASE(Int, 16);
796  WIDTH_CASE(Int, 64);
797  default:
798  llvm_unreachable("invalid bitwidth to getCapabilities");
799  }
800  } else {
801  assert(isa<FloatType>());
802  switch (bitwidth) {
803  case 32:
804  break;
805  WIDTH_CASE(Float, 16);
806  WIDTH_CASE(Float, 64);
807  default:
808  llvm_unreachable("invalid bitwidth to getCapabilities");
809  }
810  }

812 #undef WIDTH_CASE
