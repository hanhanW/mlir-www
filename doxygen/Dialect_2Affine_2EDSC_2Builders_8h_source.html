32 ///
33 /// Explicit nested LoopBuilder. Offers a compressed multi-loop builder to avoid
34 /// explicitly writing all the loops in a nest. This simple functionality is
35 /// also useful to write rank-agnostic custom ops.
36 ///
37 /// Usage:
38 ///
39 /// ```c++
40 /// AffineLoopNestBuilder({&i, &j, &k}, {lb, lb, lb}, {ub, ub, ub}, {1, 1,
41 /// 1})(
42 /// [&](){
43 /// ...
44 /// });
45 /// ```
46 ///
47 /// ```c++
48 /// AffineLoopNestBuilder({&i}, {lb}, {ub}, {1})([&](){
49 /// AffineLoopNestBuilder({&j}, {lb}, {ub}, {1})([&](){
50 /// AffineLoopNestBuilder({&k}, {lb}, {ub}, {1})([&](){
51 /// ...
52 /// }),
53 /// }),
54 /// });
55 /// ```
56 class AffineLoopNestBuilder {
57 public:
58  /// This entry point accommodates the fact that AffineForOp implicitly uses
59  /// multiple `lbs` and `ubs` with one single `iv` and `step` to encode `max`
60  /// and and `min` constraints respectively.
61  AffineLoopNestBuilder(Value *iv, ArrayRef<Value> lbs, ArrayRef<Value> ubs,
62  int64_t step);
63  AffineLoopNestBuilder(MutableArrayRef<Value> ivs, ArrayRef<Value> lbs,
64  ArrayRef<Value> ubs, ArrayRef<int64_t> steps);
65 
66  void operator()(function_ref<void(void)> fun = nullptr);
67 
68 private:
69  SmallVector<LoopBuilder, 4> loops;
70 };

72 /// Creates a perfect nest of affine "for" loops, given the list of lower
73 /// bounds, upper bounds and steps. The three lists are expected to contain the
74 /// same number of elements. Uses the OpBuilder and Location stored in
75 /// ScopedContext and assumes they are non-null. The optional "bodyBuilderFn"
76 /// callback is called to construct the body of the innermost loop and is passed
77 /// the list of loop induction variables, in order from outermost to innermost.
78 /// The function is expected to use the builder and location stored in
79 /// ScopedContext at the moment of the call. The function should not create
80 /// the affine terminator op, which will be added regardless of the
81 /// "bodyBuilderFn" being present.
82 void affineLoopNestBuilder(
83  ValueRange lbs, ValueRange ubs, ArrayRef<int64_t> steps,
84  function_ref<void(ValueRange)> bodyBuilderFn = nullptr);
85 
86 /// Creates a single affine "for" loop, iterating from max(lbs) to min(ubs) with
87 /// the given step. Uses the OpBuilder and Location stored in ScopedContext and
88 /// assumes they are non-null. The optional "bodyBuilderFn" callback is called
89 /// to construct the body of the loop and is passed the induction variable. The
90 /// function is expected to use the builder and location stored in ScopedContext
91 /// at the moment of the call. The function should not create the affine
92 /// terminator op, which will be added regardless of the "bodyBuilderFn" being
93 /// present.
94 void affineLoopBuilder(ValueRange lbs, ValueRange ubs, int64_t step,
95  function_ref<void(Value)> bodyBuilderFn = nullptr);
