17 static Optional<Value> emitStaticFor(ArrayRef<Value> lbs, ArrayRef<Value> ubs,
18  int64_t step) {
19  if (lbs.size() != 1 || ubs.size() != 1)
20  return Optional<Value>();
21 
22  auto *lbDef = lbs.front().getDefiningOp();
23  auto *ubDef = ubs.front().getDefiningOp();
24  if (!lbDef || !ubDef)
25  return Optional<Value>();
26 
27  auto lbConst = dyn_cast<ConstantIndexOp>(lbDef);
28  auto ubConst = dyn_cast<ConstantIndexOp>(ubDef);
29  if (!lbConst || !ubConst)
30  return Optional<Value>();
31  return ScopedContext::getBuilderRef()
32  .create<AffineForOp>(ScopedContext::getLocation(), lbConst.getValue(),
33  ubConst.getValue(), step)
34  .getInductionVar();
35 }

37 LoopBuilder mlir::edsc::makeAffineLoopBuilder(Value *iv, ArrayRef<Value> lbs,
38  ArrayRef<Value> ubs,
39  int64_t step) {
40  mlir::edsc::LoopBuilder result;
41  if (auto staticForIv = emitStaticFor(lbs, ubs, step))
42  *iv = staticForIv.getValue();
43  else
44  *iv = ScopedContext::getBuilderRef()
45  .create<AffineForOp>(
46  ScopedContext::getLocation(), lbs,
47  ScopedContext::getBuilderRef().getMultiDimIdentityMap(
48  lbs.size()),
49  ubs,
50  ScopedContext::getBuilderRef().getMultiDimIdentityMap(
51  ubs.size()),
52  step)
53  .getInductionVar();
54 
55  auto *body = getForInductionVarOwner(*iv).getBody();
56  result.enter(body);
57  return result;
58 }

67 void mlir::edsc::affineLoopNestBuilder(
68  ValueRange lbs, ValueRange ubs, ArrayRef<int64_t> steps,
69  function_ref<void(ValueRange)> bodyBuilderFn) {
70  assert(ScopedContext::getContext() && "EDSC ScopedContext not set up");
71  assert(lbs.size() == ubs.size() && "Mismatch in number of arguments");
72  assert(lbs.size() == steps.size() && "Mismatch in number of arguments");
73 
74  // If there are no loops to be constructed, construct the body anyway.
75  if (lbs.empty()) {
76  if (bodyBuilderFn)
77  bodyBuilderFn(ValueRange());
78  return;
79  }
80 
81  // Fetch the builder and location.
82  OpBuilder &builder = ScopedContext::getBuilderRef();
83  OpBuilder::InsertionGuard guard(builder);
84  Location loc = ScopedContext::getLocation();
85  AffineMap identity = builder.getDimIdentityMap();
86 
87  // Create the loops iteratively and store the induction variables.
88  SmallVector<Value, 4> ivs;
89  ivs.reserve(lbs.size());
90  for (unsigned i = 0, e = lbs.size(); i < e; ++i) {
91  // Callback for creating the loop body, always creates the terminator.
92  auto loopBody = [&](OpBuilder &nestedBuilder, Location nestedLoc,
93  Value iv) {
94  ivs.push_back(iv);
95  // In the innermost loop, call the body builder.
96  if (i == e - 1 && bodyBuilderFn) {
97  ScopedContext nestedContext(nestedBuilder, loc);
98  OpBuilder::InsertionGuard nestedGuard(nestedBuilder);
99  bodyBuilderFn(ivs);
100  }
101  nestedBuilder.create<AffineTerminatorOp>(nestedLoc);
102  };
103 
104  // Create the loop. If the bounds are known to be constants, use the
105  // constant form of the loop.
106  auto lbConst = lbs[i].getDefiningOp<ConstantIndexOp>();
107  auto ubConst = ubs[i].getDefiningOp<ConstantIndexOp>();
108  auto loop = lbConst && ubConst
109  ? builder.create<AffineForOp>(loc, lbConst.getValue(),
110  ubConst.getValue(), steps[i],
111  loopBody)
112  : builder.create<AffineForOp>(loc, lbs[i], identity, ubs[i],
113  identity, steps[i], loopBody);
114  builder.setInsertionPointToStart(loop.getBody());
115  }
116 }

118 void mlir::edsc::affineLoopBuilder(ValueRange lbs, ValueRange ubs, int64_t step,
119  function_ref<void(Value)> bodyBuilderFn) {
120  // Fetch the builder and location.
121  assert(ScopedContext::getContext() && "EDSC ScopedContext not set up");
122  OpBuilder &builder = ScopedContext::getBuilderRef();
123  Location loc = ScopedContext::getLocation();
124 
125  // Create the actual loop and call the body builder, if provided, after
126  // updating the scoped context.
127  builder.create<AffineForOp>(
128  loc, lbs, builder.getMultiDimIdentityMap(lbs.size()), ubs,
129  builder.getMultiDimIdentityMap(ubs.size()), step,
130  [&](OpBuilder &nestedBuilder, Location nestedLoc, Value iv) {
131  if (bodyBuilderFn) {
132  ScopedContext nestedContext(nestedBuilder, nestedLoc);
133  OpBuilder::InsertionGuard guard(nestedBuilder);
134  bodyBuilderFn(iv);
135  }
136  nestedBuilder.create<AffineTerminatorOp>(nestedLoc);
137  });
138 }

140 mlir::edsc::AffineLoopNestBuilder::AffineLoopNestBuilder(
141  MutableArrayRef<Value> ivs, ArrayRef<Value> lbs, ArrayRef<Value> ubs,
142  ArrayRef<int64_t> steps) {
143  assert(ivs.size() == lbs.size() && "Mismatch in number of arguments");
144  assert(ivs.size() == ubs.size() && "Mismatch in number of arguments");
145  assert(ivs.size() == steps.size() && "Mismatch in number of arguments");
146  for (auto it : llvm::zip(ivs, lbs, ubs, steps))
147  loops.emplace_back(makeAffineLoopBuilder(&std::get<0>(it), std::get<1>(it),
148  std::get<2>(it), std::get<3>(it)));
149 }

151 void mlir::edsc::AffineLoopNestBuilder::operator()(
152  function_ref<void(void)> fun) {
153  if (fun)
154  fun();
155  // Iterate on the calling operator() on all the loops in the nest.
156  // The iteration order is from innermost to outermost because enter/exit needs
157  // to be asymmetric (i.e. enter() occurs on LoopBuilder construction, exit()
158  // occurs on calling operator()). The asymmetry is required for properly
159  // nesting imperfectly nested regions (see LoopBuilder::operator()).
160  for (auto lit = loops.rbegin(), eit = loops.rend(); lit != eit; ++lit)
161  (*lit)();
162 }

164 static std::pair<AffineExpr, Value>
165 categorizeValueByAffineType(MLIRContext *context, Value val, unsigned &numDims,
166  unsigned &numSymbols) {
167  AffineExpr d;
168  Value resultVal = nullptr;
169  if (auto constant = val.getDefiningOp<ConstantIndexOp>()) {
170  d = getAffineConstantExpr(constant.getValue(), context);
171  } else if (isValidSymbol(val) && !isValidDim(val)) {
172  d = getAffineSymbolExpr(numSymbols++, context);
173  resultVal = val;
174  } else {
175  d = getAffineDimExpr(numDims++, context);
176  resultVal = val;
177  }
178  return std::make_pair(d, resultVal);
179 }

181 static Value createBinaryIndexHandle(
182  Value lhs, Value rhs,
183  function_ref<AffineExpr(AffineExpr, AffineExpr)> affCombiner) {
184  MLIRContext *context = ScopedContext::getContext();
185  unsigned numDims = 0, numSymbols = 0;
186  AffineExpr d0, d1;
187  Value v0, v1;
188  std::tie(d0, v0) =
189  categorizeValueByAffineType(context, lhs, numDims, numSymbols);
190  std::tie(d1, v1) =
191  categorizeValueByAffineType(context, rhs, numDims, numSymbols);
192  SmallVector<Value, 2> operands;
193  if (v0)
194  operands.push_back(v0);
195  if (v1)
196  operands.push_back(v1);
197  auto map = AffineMap::get(numDims, numSymbols, affCombiner(d0, d1));
198 
199  // TODO: createOrFold when available.
200  Operation *op =
201  makeComposedAffineApply(ScopedContext::getBuilderRef(),
202  ScopedContext::getLocation(), map, operands)
203  .getOperation();
204  assert(op->getNumResults() == 1 && "Expected single result AffineApply");
205  return op->getResult(0);
206 }

208 template <typename IOp, typename FOp>
209 static Value createBinaryHandle(
210  Value lhs, Value rhs,
211  function_ref<AffineExpr(AffineExpr, AffineExpr)> affCombiner) {
212  auto thisType = lhs.getType();
213  auto thatType = rhs.getType();
214  assert(thisType == thatType && "cannot mix types in operators");
215  (void)thisType;
216  (void)thatType;
217  if (thisType.isIndex()) {
218  return createBinaryIndexHandle(lhs, rhs, affCombiner);
219  } else if (thisType.isSignlessInteger()) {
220  return ValueBuilder<IOp>(lhs, rhs);
221  } else if (thisType.isa<FloatType>()) {
222  return ValueBuilder<FOp>(lhs, rhs);
223  } else if (thisType.isa<VectorType>() || thisType.isa<TensorType>()) {
224  auto aggregateType = thisType.cast<ShapedType>();
225  if (aggregateType.getElementType().isSignlessInteger())
226  return ValueBuilder<IOp>(lhs, rhs);
227  else if (aggregateType.getElementType().isa<FloatType>())
228  return ValueBuilder<FOp>(lhs, rhs);
229  }
230  llvm_unreachable("failed to create a Value");
231 }
