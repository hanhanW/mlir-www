31 /// Base storage class appearing in a Type.
32 class TypeStorage : public StorageUniquer::BaseStorage {
33  friend detail::TypeUniquer;

36 protected:
37  /// This constructor is used by derived classes as part of the TypeUniquer.

39  /// invoked afterwards for the storage to be valid.
40  TypeStorage(unsigned subclassData = 0)
41  : dialect(nullptr), subclassData(subclassData) {}
42 
43 public:
44  /// Get the dialect that this type is registered to.
45  Dialect &getDialect() {
46  assert(dialect && "Malformed type storage object.");

48  }
49  /// Get the subclass data.
50  unsigned getSubclassData() const { return subclassData; }

55 private:

60  /// The dialect for this type.

82 namespace detail {
83 // A utility class to get, or create, unique instances of types within an
84 // MLIRContext. This class manages all creation and uniquing of types.
85 class TypeUniquer {
86 public:
87  /// Get an uniqued instance of a type T.
88  template <typename T, typename... Args>
89  static T get(MLIRContext *ctx, unsigned kind, Args &&... args) {
90  return ctx->getTypeUniquer().get<typename T::ImplType>(
91  [&](TypeStorage *storage) {
92  storage->initializeDialect(lookupDialectForType<T>(ctx));
93  },
94  kind, std::forward<Args>(args)...);
95  }
96 
97 private:
98  /// Get the dialect that the type 'T' was registered with.
99  template <typename T> static Dialect &lookupDialectForType(MLIRContext *ctx) {
100  return lookupDialectForType(ctx, T::getTypeID());
101  }
102 
103  /// Get the dialect that registered the type with the provided typeid.
104  static Dialect &lookupDialectForType(MLIRContext *ctx, TypeID typeID);
105 };
