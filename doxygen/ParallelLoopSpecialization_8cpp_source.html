24 /// Rewrite a loop with bounds defined by an affine.min with a constant into 2
25 /// loops after checking if the bounds are equal to that constant. This is
26 /// beneficial if the loop will almost always have the constant bound and that
27 /// version can be fully unrolled and vectorized.
28 static void specializeLoopForUnrolling(ParallelOp op) {
29  SmallVector<int64_t, 2> constantIndices;
30  constantIndices.reserve(op.upperBound().size());
31  for (auto bound : op.upperBound()) {
32  auto minOp = bound.getDefiningOp<AffineMinOp>();
33  if (!minOp)
34  return;
35  int64_t minConstant = std::numeric_limits<int64_t>::max();
36  for (auto expr : minOp.map().getResults()) {
37  if (auto constantIndex = expr.dyn_cast<AffineConstantExpr>())
38  minConstant = std::min(minConstant, constantIndex.getValue());
39  }
40  if (minConstant == std::numeric_limits<int64_t>::max())
41  return;
42  constantIndices.push_back(minConstant);
43  }
44 
45  OpBuilder b(op);
46  BlockAndValueMapping map;
47  Value cond;
48  for (auto bound : llvm::zip(op.upperBound(), constantIndices)) {
49  Value constant = b.create<ConstantIndexOp>(op.getLoc(), std::get<1>(bound));
50  Value cmp = b.create<CmpIOp>(op.getLoc(), CmpIPredicate::eq,
51  std::get<0>(bound), constant);
52  cond = cond ? b.create<AndOp>(op.getLoc(), cond, cmp) : cmp;
53  map.map(std::get<0>(bound), constant);
54  }
55  auto ifOp = b.create<scf::IfOp>(op.getLoc(), cond, /*withElseRegion=*/true);
56  ifOp.getThenBodyBuilder().clone(*op.getOperation(), map);
57  ifOp.getElseBodyBuilder().clone(*op.getOperation());
58  op.erase();
59 }
