94 /// A NestedBuilder is a scoping abstraction to create an idiomatic syntax
95 /// embedded in C++ that serves the purpose of building nested MLIR.
96 /// Nesting and compositionality is obtained by using the strict ordering that
97 /// exists between object construction and method invocation on said object (in
98 /// our case, the call to `operator()`).
99 /// This ordering allows implementing an abstraction that decouples definition
100 /// from declaration (in a PL sense) on placeholders.
101 class NestedBuilder {
102 protected:
103  NestedBuilder() = default;
104  NestedBuilder(const NestedBuilder &) = delete;
105  NestedBuilder(NestedBuilder &&other) : bodyScope(other.bodyScope) {
106  other.bodyScope = nullptr;
107  }
108 
109  NestedBuilder &operator=(const NestedBuilder &) = delete;
110  NestedBuilder &operator=(NestedBuilder &&other) {
111  std::swap(bodyScope, other.bodyScope);
112  return *this;
113  }

115  /// Enter an mlir::Block and setup a ScopedContext to insert operations at
116  /// the end of it. Since we cannot use c++ language-level scoping to implement
117  /// scoping itself, we use enter/exit pairs of operations.
118  /// As a consequence we must allocate a new OpBuilder + ScopedContext and
119  /// let the escape.
120  void enter(mlir::Block *block) {
121  bodyScope = new ScopedContext(ScopedContext::getBuilderRef(),
122  OpBuilder::InsertPoint(block, block->end()),
123  ScopedContext::getLocation());
124  if (!block->empty()) {
125  auto &termOp = block->back();
126  if (termOp.isKnownTerminator())
127  ScopedContext::getBuilderRef().setInsertionPoint(&termOp);
128  }
129  }
130 
131  /// Exit the current mlir::Block by explicitly deleting the dynamically
132  /// allocated OpBuilder and ScopedContext.
133  void exit() {
134  delete bodyScope;
135  bodyScope = nullptr;
136  }
137 
138  /// Custom destructor does nothing because we already destroyed bodyScope
139  /// manually in `exit`. Insert an assertion to defensively guard against
140  /// improper usage of scoping.
141  ~NestedBuilder() {
142  assert(!bodyScope &&
143  "Illegal use of NestedBuilder; must have called exit()");
144  }
145 
146 private:
147  ScopedContext *bodyScope = nullptr;
148 };

272 /// A StructuredIndexed represents an indexable quantity that is either:
273 /// 1. a captured value, which is suitable for buffer and tensor operands, or;
274 /// 2. a captured type, which is suitable for tensor return values.
275 ///
276 /// A StructuredIndexed itself is indexed and passed to `makeGenericLinalgOp`.
277 /// It enable an idiomatic syntax for index expressions such as:
278 ///
279 /// ```
280 /// StructuredIndexed A(buffer_or_tensor_value), B(buffer_or_tensor_value),
281 /// C(buffer_value_or_tensor_type);
282 /// makeGenericLinalgOp({A({m, n}), B({k, n})}, {C({m, n})}, ... );
283 /// ```
284 struct StructuredIndexed {
285  StructuredIndexed(Value v) : value(v) {}
286  StructuredIndexed(Type t) : type(t) {}
287  StructuredIndexed operator()(ArrayRef<AffineExpr> indexings) {
288  return value ? StructuredIndexed(value, indexings)
289  : StructuredIndexed(type, indexings);
290  }
291 
292  StructuredIndexed(Value v, ArrayRef<AffineExpr> indexings)
293  : value(v), exprs(indexings.begin(), indexings.end()) {
294  assert((v.getType().isa<MemRefType>() ||
295  v.getType().isa<RankedTensorType>() ||
296  v.getType().isa<VectorType>()) &&
297  "MemRef, RankedTensor or Vector expected");
298  }
299  StructuredIndexed(Type t, ArrayRef<AffineExpr> indexings)
300  : type(t), exprs(indexings.begin(), indexings.end()) {
301  assert((t.isa<MemRefType>() || t.isa<RankedTensorType>() ||
302  t.isa<VectorType>()) &&
303  "MemRef, RankedTensor or Vector expected");
304  }
305 
306  bool hasValue() const { return (bool)value; }
307  Value getValue() const {
308  assert(value && "StructuredIndexed Value not set.");
309  return value;
310  }

312  assert((value || type) && "StructuredIndexed Value and Type not set.");
313  return value ? value.getType() : type;
314  }

316  operator Value() const { return getValue(); }
