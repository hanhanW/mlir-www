47 /// Given the 'indices' of an load/store operation where the memref is a result
48 /// of a subview op, returns the indices w.r.t to the source memref of the
49 /// subview op. For example
50 ///
51 /// %0 = ... : memref<12x42xf32>
52 /// %1 = subview %0[%arg0, %arg1][][%stride1, %stride2] : memref<12x42xf32> to
53 /// memref<4x4xf32, offset=?, strides=[?, ?]>
54 /// %2 = load %1[%i1, %i2] : memref<4x4xf32, offset=?, strides=[?, ?]>
55 ///
56 /// could be folded into
57 ///
58 /// %2 = load %0[%arg0 + %i1 * %stride1][%arg1 + %i2 * %stride2] :
59 /// memref<12x42xf32>
60 static LogicalResult
61 resolveSourceIndices(Location loc, PatternRewriter &rewriter,
62  SubViewOp subViewOp, ValueRange indices,
63  SmallVectorImpl<Value> &sourceIndices) {
64  // TODO: Aborting when the offsets are static. There might be a way to fold
65  // the subview op with load even if the offsets have been canonicalized
66  // away.
67  SmallVector<Value, 4> opOffsets = subViewOp.getOrCreateOffsets(rewriter, loc);
68  SmallVector<Value, 4> opStrides = subViewOp.getOrCreateStrides(rewriter, loc);
69  assert(opOffsets.size() == indices.size() &&
70  "expected as many indices as rank of subview op result type");
71  assert(opStrides.size() == indices.size() &&
72  "expected as many indices as rank of subview op result type");
73 
74  // New indices for the load are the current indices * subview_stride +
75  // subview_offset.
76  sourceIndices.resize(indices.size());
77  for (auto index : llvm::enumerate(indices)) {
78  auto offset = opOffsets[index.index()];
79  auto stride = opStrides[index.index()];
80  auto mul = rewriter.create<MulIOp>(loc, index.value(), stride);
81  sourceIndices[index.index()] =
82  rewriter.create<AddIOp>(loc, offset, mul).getResult();
83  }
84  return success();
85 }

91 LogicalResult
92 LoadOpOfSubViewFolder::matchAndRewrite(LoadOp loadOp,
93  PatternRewriter &rewriter) const {
94  auto subViewOp = loadOp.memref().getDefiningOp<SubViewOp>();
95  if (!subViewOp) {
96  return failure();
97  }
98  SmallVector<Value, 4> sourceIndices;
99  if (failed(resolveSourceIndices(loadOp.getLoc(), rewriter, subViewOp,
100  loadOp.indices(), sourceIndices)))
101  return failure();
102 
103  rewriter.replaceOpWithNewOp<LoadOp>(loadOp, subViewOp.source(),
104  sourceIndices);
105  return success();
106 }
