53 /// Integer Type Storage and Uniquing.
54 struct IntegerTypeStorage : public TypeStorage {
55  IntegerTypeStorage(unsigned width,
56  IntegerType::SignednessSemantics signedness)
57  : TypeStorage(packKeyBits(width, signedness)) {}
58 
59  /// The hash key used for uniquing.
60  using KeyTy = std::pair<unsigned, IntegerType::SignednessSemantics>;
61 
62  static llvm::hash_code hashKey(const KeyTy &key) {
63  return llvm::hash_value(packKeyBits(key.first, key.second));
64  }
65 
66  bool operator==(const KeyTy &key) const {
67  return getSubclassData() == packKeyBits(key.first, key.second);
68  }
69 
70  static IntegerTypeStorage *construct(TypeStorageAllocator &allocator,
71  KeyTy key) {
72  return new (allocator.allocate<IntegerTypeStorage>())
73  IntegerTypeStorage(key.first, key.second);
74  }

76  struct KeyBits {
77  unsigned width : 30;
78  unsigned signedness : 2;
79  };
80 
81  /// Pack the given `width` and `signedness` as a key.
82  static unsigned packKeyBits(unsigned width,
83  IntegerType::SignednessSemantics signedness) {
84  KeyBits bits{width, static_cast<unsigned>(signedness)};
85  return llvm::bit_cast<unsigned>(bits);
86  }
87 
88  static KeyBits unpackKeyBits(unsigned bits) {
89  return llvm::bit_cast<KeyBits>(bits);
90  }
91 
92  unsigned getWidth() { return unpackKeyBits(getSubclassData()).width; }
93 
94  IntegerType::SignednessSemantics getSignedness() {
95  return static_cast<IntegerType::SignednessSemantics>(
96  unpackKeyBits(getSubclassData()).signedness);
97  }
98 };

100 /// Function Type Storage and Uniquing.
101 struct FunctionTypeStorage : public TypeStorage {
102  FunctionTypeStorage(unsigned numInputs, unsigned numResults,
103  Type const *inputsAndResults)
104  : TypeStorage(numInputs), numResults(numResults),
105  inputsAndResults(inputsAndResults) {}
106 
107  /// The hash key used for uniquing.
108  using KeyTy = std::pair<ArrayRef<Type>, ArrayRef<Type>>;
109  bool operator==(const KeyTy &key) const {
110  return key == KeyTy(getInputs(), getResults());
111  }
112 
113  /// Construction.
114  static FunctionTypeStorage *construct(TypeStorageAllocator &allocator,
115  const KeyTy &key) {
116  ArrayRef<Type> inputs = key.first, results = key.second;
117 
118  // Copy the inputs and results into the bump pointer.
119  SmallVector<Type, 16> types;
120  types.reserve(inputs.size() + results.size());
121  types.append(inputs.begin(), inputs.end());
122  types.append(results.begin(), results.end());
123  auto typesList = allocator.copyInto(ArrayRef<Type>(types));
124 
125  // Initialize the memory using placement new.
126  return new (allocator.allocate<FunctionTypeStorage>())
127  FunctionTypeStorage(inputs.size(), results.size(), typesList.data());
128  }
129 
130  ArrayRef<Type> getInputs() const {
131  return ArrayRef<Type>(inputsAndResults, getSubclassData());
132  }
133  ArrayRef<Type> getResults() const {
134  return ArrayRef<Type>(inputsAndResults + getSubclassData(), numResults);
135  }
136 
137  unsigned numResults;
138  Type const *inputsAndResults;
139 };

224 struct MemRefTypeStorage : public ShapedTypeStorage {
225  MemRefTypeStorage(unsigned shapeSize, Type elementType,
226  const int64_t *shapeElements, const unsigned numAffineMaps,
227  AffineMap const *affineMapList, const unsigned memorySpace)
228  : ShapedTypeStorage(elementType, shapeSize), shapeElements(shapeElements),
229  numAffineMaps(numAffineMaps), affineMapList(affineMapList),
230  memorySpace(memorySpace) {}
231 
232  /// The hash key used for uniquing.
233  // MemRefs are uniqued based on their shape, element type, affine map
234  // composition, and memory space.
235  using KeyTy =
236  std::tuple<ArrayRef<int64_t>, Type, ArrayRef<AffineMap>, unsigned>;
237  bool operator==(const KeyTy &key) const {
238  return key == KeyTy(getShape(), elementType, getAffineMaps(), memorySpace);
239  }
240 
241  /// Construction.
242  static MemRefTypeStorage *construct(TypeStorageAllocator &allocator,
243  const KeyTy &key) {
244  // Copy the shape into the bump pointer.
245  ArrayRef<int64_t> shape = allocator.copyInto(std::get<0>(key));
246 
247  // Copy the affine map composition into the bump pointer.
248  ArrayRef<AffineMap> affineMapComposition =
249  allocator.copyInto(std::get<2>(key));
250 
251  // Initialize the memory using placement new.
252  return new (allocator.allocate<MemRefTypeStorage>())
253  MemRefTypeStorage(shape.size(), std::get<1>(key), shape.data(),
254  affineMapComposition.size(),
255  affineMapComposition.data(), std::get<3>(key));
256  }
257 
258  ArrayRef<int64_t> getShape() const {
259  return ArrayRef<int64_t>(shapeElements, getSubclassData());
260  }
261 
262  ArrayRef<AffineMap> getAffineMaps() const {
263  return ArrayRef<AffineMap>(affineMapList, numAffineMaps);
264  }
265 
266  /// An array of integers which stores the shape dimension sizes.
267  const int64_t *shapeElements;
268  /// The number of affine maps in the 'affineMapList' array.
269  const unsigned numAffineMaps;
270  /// List of affine maps in the memref's layout/index map composition.
271  AffineMap const *affineMapList;
272  /// Memory space in which data referenced by memref resides.
273  const unsigned memorySpace;
274 };

319 /// A type representing a collection of other types.
320 struct TupleTypeStorage final
321  : public TypeStorage,
322  public llvm::TrailingObjects<TupleTypeStorage, Type> {
323  using KeyTy = ArrayRef<Type>;
324 
325  TupleTypeStorage(unsigned numTypes) : TypeStorage(numTypes) {}
326 
327  /// Construction.
328  static TupleTypeStorage *construct(TypeStorageAllocator &allocator,
329  ArrayRef<Type> key) {
330  // Allocate a new storage instance.
331  auto byteSize = TupleTypeStorage::totalSizeToAlloc<Type>(key.size());
332  auto rawMem = allocator.allocate(byteSize, alignof(TupleTypeStorage));
333  auto result = ::new (rawMem) TupleTypeStorage(key.size());
334 
335  // Copy in the element types into the trailing storage.
336  std::uninitialized_copy(key.begin(), key.end(),
337  result->getTrailingObjects<Type>());
338  return result;
339  }
340 
341  bool operator==(const KeyTy &key) const { return key == getTypes(); }
342 
343  /// Return the number of held types.
344  unsigned size() const { return getSubclassData(); }
345 
346  /// Return the held types.
347  ArrayRef<Type> getTypes() const {
348  return {getTrailingObjects<Type>(), size()};
349  }
