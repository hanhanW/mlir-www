32 LogicalResult
33 ReduceOpConverter::matchAndRewrite(ReduceOp reduceOp,
34  PatternRewriter &rewriter) const {
35  auto loc = reduceOp.getLoc();
36 
37  Value zero = rewriter.create<ConstantIndexOp>(loc, 0);
38  Value one = rewriter.create<ConstantIndexOp>(loc, 1);
39  Value extentTensor = rewriter.create<ToExtentTensorOp>(
40  loc,
41  RankedTensorType::get({ShapedType::kDynamicSize},
42  rewriter.getIndexType()),
43  reduceOp.shape());
44  Value size =
45  rewriter.create<DimOp>(loc, rewriter.getIndexType(), extentTensor, zero);
46 
47  auto loop = rewriter.create<scf::ForOp>(
48  loc, zero, size, one, reduceOp.initVals(),
49  [&](OpBuilder &b, Location nestedLoc, Value iv, ValueRange args) {
50  Value indexExtent = b.create<ExtractElementOp>(loc, extentTensor, iv);
51  Value sizeExtent = b.create<IndexToSizeOp>(loc, indexExtent);
52 
53  SmallVector<Value, 2> mapped_values{iv, sizeExtent};
54  mapped_values.append(args.begin(), args.end());
55 
56  BlockAndValueMapping mapping;
57  Block *reduceBody = reduceOp.getBody();
58  mapping.map(reduceBody->getArguments(), mapped_values);
59  for (auto &nested : reduceBody->without_terminator())
60  b.clone(nested, mapping);
61 
62  SmallVector<Value, 2> mappedResults;
63  for (auto result : reduceBody->getTerminator()->getOperands())
64  mappedResults.push_back(mapping.lookup(result));
65  b.create<scf::YieldOp>(loc, mappedResults);
66  });
67 
68  rewriter.replaceOp(reduceOp, loop.getResults());
69  return success();
70 }
