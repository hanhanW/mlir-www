27 void mlir::edsc::ParallelLoopNestBuilder::operator()(
28  function_ref<void(void)> fun) {
29  if (fun)
30  fun();
31  // Iterate on the calling operator() on all the loops in the nest.
32  // The iteration order is from innermost to outermost because enter/exit needs
33  // to be asymmetric (i.e. enter() occurs on LoopBuilder construction, exit()
34  // occurs on calling operator()). The asymmetry is required for properly
35  // nesting imperfectly nested regions (see LoopBuilder::operator()).
36  for (auto lit = loops.rbegin(), eit = loops.rend(); lit != eit; ++lit)
37  (*lit)();
38 }

40 mlir::edsc::LoopNestBuilder::LoopNestBuilder(MutableArrayRef<Value> ivs,
41  ArrayRef<Value> lbs,
42  ArrayRef<Value> ubs,
43  ArrayRef<Value> steps) {
44  assert(ivs.size() == lbs.size() && "expected size of ivs and lbs to match");
45  assert(ivs.size() == ubs.size() && "expected size of ivs and ubs to match");
46  assert(ivs.size() == steps.size() &&
47  "expected size of ivs and steps to match");
48  loops.reserve(ivs.size());
49  for (auto it : llvm::zip(ivs, lbs, ubs, steps))
50  loops.emplace_back(makeLoopBuilder(&std::get<0>(it), std::get<1>(it),
51  std::get<2>(it), std::get<3>(it)));
52  assert(loops.size() == ivs.size() && "Mismatch loops vs ivs size");
53 }

83 LoopBuilder mlir::edsc::makeParallelLoopBuilder(MutableArrayRef<Value> ivs,
84  ArrayRef<Value> lbs,
85  ArrayRef<Value> ubs,
86  ArrayRef<Value> steps) {
87  scf::ParallelOp parallelOp = OperationBuilder<scf::ParallelOp>(
88  SmallVector<Value, 4>(lbs.begin(), lbs.end()),
89  SmallVector<Value, 4>(ubs.begin(), ubs.end()),
90  SmallVector<Value, 4>(steps.begin(), steps.end()));
91  for (size_t i = 0, e = ivs.size(); i < e; ++i)
92  ivs[i] = parallelOp.getBody()->getArgument(i);
93  LoopBuilder result;
94  result.enter(parallelOp.getBody());
95  return result;
96 }

98 mlir::edsc::LoopBuilder
99 mlir::edsc::makeLoopBuilder(Value *iv, Value lb, Value ub, Value step,
100  MutableArrayRef<Value> iterArgsHandles,
101  ValueRange iterArgsInitValues) {
102  mlir::edsc::LoopBuilder result;
103  scf::ForOp forOp =
104  OperationBuilder<scf::ForOp>(lb, ub, step, iterArgsInitValues);
105  *iv = Value(forOp.getInductionVar());
106  auto *body = scf::getForInductionVarOwner(*iv).getBody();
107  for (size_t i = 0, e = iterArgsHandles.size(); i < e; ++i) {
108  // Skipping the induction variable.
109  iterArgsHandles[i] = body->getArgument(i + 1);
110  }
111  result.setOp(forOp);
112  result.enter(body);
113  return result;
114 }

163 static std::function<void(OpBuilder &, Location)>
164 wrapIfBody(function_ref<scf::ValueVector()> body, TypeRange expectedTypes) {
165  (void)expectedTypes;
166  return [=](OpBuilder &builder, Location loc) {
