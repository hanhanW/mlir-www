47 /// Enum to control the lowering of `vector.contract` operations.
48 enum class VectorContractLowering {
49  /// Progressively lower to finer grained `vector.contract` and `vector.fma`.
50  FMA = 0,
51  /// Lower to `vector.matrix_multiply`, maps 1-1 to LLVM matrix intrinsics.
52  Matmul = 1,
53  /// Lower to `vector.outerproduct`.
54  OuterProduct = 2,
55 };
56 /// Enum to control the lowering of `vector.transpose` operations.
57 enum class VectorTransposeLowering {
58  // Lower transpose into element-wise extract and inserts.
59  EltWise = 0,
60  /// Lower 2-D transpose to `vector.flat_transpose`, maps 1-1 to LLVM matrix
61  /// intrinsics.
62  Flat = 1,
63 };
