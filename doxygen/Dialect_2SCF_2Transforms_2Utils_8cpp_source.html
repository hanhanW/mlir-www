20 scf::ForOp mlir::cloneWithNewYields(OpBuilder &b, scf::ForOp loop,
21  ValueRange newIterOperands,
22  ValueRange newYieldedValues,
23  bool replaceLoopResults) {
24  assert(newIterOperands.size() == newYieldedValues.size() &&
25  "newIterOperands must be of the same size as newYieldedValues");
26 
27  // Create a new loop before the existing one, with the extra operands.
28  OpBuilder::InsertionGuard g(b);
29  b.setInsertionPoint(loop);
30  auto operands = llvm::to_vector<4>(loop.getIterOperands());
31  operands.append(newIterOperands.begin(), newIterOperands.end());
32  scf::ForOp newLoop =
33  b.create<scf::ForOp>(loop.getLoc(), loop.lowerBound(), loop.upperBound(),
34  loop.step(), operands);
35 
36  auto &loopBody = *loop.getBody();
37  auto &newLoopBody = *newLoop.getBody();
38  // Clone / erase the yield inside the original loop to both:
39  // 1. augment its operands with the newYieldedValues.
40  // 2. automatically apply the BlockAndValueMapping on its operand
41  auto yield = cast<scf::YieldOp>(loopBody.getTerminator());
42  b.setInsertionPoint(yield);
43  auto yieldOperands = llvm::to_vector<4>(yield.getOperands());
44  yieldOperands.append(newYieldedValues.begin(), newYieldedValues.end());
45  auto newYield = b.create<scf::YieldOp>(yield.getLoc(), yieldOperands);
46 
47  // Clone the loop body with remaps.
48  BlockAndValueMapping bvm;
49  // a. remap the induction variable.
50  bvm.map(loop.getInductionVar(), newLoop.getInductionVar());
51  // b. remap the BB args.
52  bvm.map(loopBody.getArguments(),
53  newLoopBody.getArguments().take_front(loopBody.getNumArguments()));
54  // c. remap the iter args.
55  bvm.map(newIterOperands,
56  newLoop.getRegionIterArgs().take_back(newIterOperands.size()));
57  b.setInsertionPointToStart(&newLoopBody);
58  // Skip the original yield terminator which does not have enough operands.
59  for (auto &o : loopBody.without_terminator())
60  b.clone(o, bvm);
61 
62  // Replace `loop`'s results if requested.
63  if (replaceLoopResults) {
64  for (auto it : llvm::zip(loop.getResults(), newLoop.getResults().take_front(
65  loop.getNumResults())))
66  std::get<0>(it).replaceAllUsesWith(std::get<1>(it));
67  }
68 
69  // TODO: this is unsafe in the context of a PatternRewrite.
70  newYield.erase();
71 
72  return newLoop;
73 }
