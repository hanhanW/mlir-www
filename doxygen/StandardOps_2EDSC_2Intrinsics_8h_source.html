56 /// Creates a new mlir::Block* and branches to it from the current block.
57 /// Argument types are specified by `operands`.
58 /// Captures the new block in `bh` and the actual `operands` in `captures`. To
59 /// insert the new mlir::Block*, a local ScopedContext is constructed and
60 /// released to the current block. The branch operation is then added to the
61 /// new block.
62 ///
63 /// Prerequisites:
64 /// `b` has not yet captured an mlir::Block*.
65 /// No `captures` have captured any mlir::Value.
66 /// All `operands` have already captured an mlir::Value
67 /// captures.size() == operands.size()
68 /// captures and operands are pairwise of the same type.
69 BranchOp std_br(BlockHandle *bh, ArrayRef<Type> types,
70  MutableArrayRef<Value> captures, ValueRange operands);
