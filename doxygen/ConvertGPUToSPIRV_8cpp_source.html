135 LogicalResult
136 ForOpConversion::matchAndRewrite(scf::ForOp forOp, ArrayRef<Value> operands,
137  ConversionPatternRewriter &rewriter) const {
138  // scf::ForOp can be lowered to the structured control flow represented by
139  // spirv::LoopOp by making the continue block of the spirv::LoopOp the loop
140  // latch and the merge block the exit block. The resulting spirv::LoopOp has a
141  // single back edge from the continue to header block, and a single exit from
142  // header to merge.
143  scf::ForOpAdaptor forOperands(operands);
144  auto loc = forOp.getLoc();
145  auto loopControl = rewriter.getI32IntegerAttr(
146  static_cast<uint32_t>(spirv::LoopControl::None));
147  auto loopOp = rewriter.create<spirv::LoopOp>(loc, loopControl);
148  loopOp.addEntryAndMergeBlock();
149 
150  OpBuilder::InsertionGuard guard(rewriter);
151  // Create the block for the header.
152  auto header = new Block();
153  // Insert the header.
154  loopOp.body().getBlocks().insert(std::next(loopOp.body().begin(), 1), header);
155 
156  // Create the new induction variable to use.
157  BlockArgument newIndVar =
158  header->addArgument(forOperands.lowerBound().getType());
159  Block *body = forOp.getBody();
160 
161  // Apply signature conversion to the body of the forOp. It has a single block,
162  // with argument which is the induction variable. That has to be replaced with
163  // the new induction variable.
164  TypeConverter::SignatureConversion signatureConverter(
165  body->getNumArguments());
166  signatureConverter.remapInput(0, newIndVar);
167  body = rewriter.applySignatureConversion(&forOp.getLoopBody(),
168  signatureConverter);
169 
170  // Delete the loop terminator.
171  rewriter.eraseOp(body->getTerminator());
172 
173  // Move the blocks from the forOp into the loopOp. This is the body of the
174  // loopOp.
175  rewriter.inlineRegionBefore(forOp.getOperation()->getRegion(0), loopOp.body(),
176  std::next(loopOp.body().begin(), 2));
177 
178  // Branch into it from the entry.
179  rewriter.setInsertionPointToEnd(&(loopOp.body().front()));
180  rewriter.create<spirv::BranchOp>(loc, header, forOperands.lowerBound());
181 
182  // Generate the rest of the loop header.
183  rewriter.setInsertionPointToEnd(header);
184  auto mergeBlock = loopOp.getMergeBlock();
185  auto cmpOp = rewriter.create<spirv::SLessThanOp>(
186  loc, rewriter.getI1Type(), newIndVar, forOperands.upperBound());
187  rewriter.create<spirv::BranchConditionalOp>(
188  loc, cmpOp, body, ArrayRef<Value>(), mergeBlock, ArrayRef<Value>());
189 
190  // Generate instructions to increment the step of the induction variable and
191  // branch to the header.
192  Block *continueBlock = loopOp.getContinueBlock();
193  rewriter.setInsertionPointToEnd(continueBlock);
194 
195  // Add the step to the induction variable and branch to the header.
196  Value updatedIndVar = rewriter.create<spirv::IAddOp>(
197  loc, newIndVar.getType(), newIndVar, forOperands.step());
198  rewriter.create<spirv::BranchOp>(loc, header, updatedIndVar);
199 
200  rewriter.eraseOp(forOp);
201  return success();
202 }

208 LogicalResult
209 IfOpConversion::matchAndRewrite(scf::IfOp ifOp, ArrayRef<Value> operands,
210  ConversionPatternRewriter &rewriter) const {
211  // When lowering `scf::IfOp` we explicitly create a selection header block
212  // before the control flow diverges and a merge block where control flow
213  // subsequently converges.
214  scf::IfOpAdaptor ifOperands(operands);
215  auto loc = ifOp.getLoc();
216 
217  // Create `spv.selection` operation, selection header block and merge block.
218  auto selectionControl = rewriter.getI32IntegerAttr(
219  static_cast<uint32_t>(spirv::SelectionControl::None));
220  auto selectionOp = rewriter.create<spirv::SelectionOp>(loc, selectionControl);
221  selectionOp.addMergeBlock();
222  auto *mergeBlock = selectionOp.getMergeBlock();
223 
224  OpBuilder::InsertionGuard guard(rewriter);
225  auto *selectionHeaderBlock = new Block();
226  selectionOp.body().getBlocks().push_front(selectionHeaderBlock);
227 
228  // Inline `then` region before the merge block and branch to it.
229  auto &thenRegion = ifOp.thenRegion();
230  auto *thenBlock = &thenRegion.front();
231  rewriter.setInsertionPointToEnd(&thenRegion.back());
232  rewriter.create<spirv::BranchOp>(loc, mergeBlock);
233  rewriter.inlineRegionBefore(thenRegion, mergeBlock);
234 
235  auto *elseBlock = mergeBlock;
236  // If `else` region is not empty, inline that region before the merge block
237  // and branch to it.
238  if (!ifOp.elseRegion().empty()) {
239  auto &elseRegion = ifOp.elseRegion();
240  elseBlock = &elseRegion.front();
241  rewriter.setInsertionPointToEnd(&elseRegion.back());
242  rewriter.create<spirv::BranchOp>(loc, mergeBlock);
243  rewriter.inlineRegionBefore(elseRegion, mergeBlock);
244  }
245 
246  // Create a `spv.BranchConditional` operation for selection header block.
247  rewriter.setInsertionPointToEnd(selectionHeaderBlock);
248  rewriter.create<spirv::BranchConditionalOp>(loc, ifOperands.condition(),
249  thenBlock, ArrayRef<Value>(),
250  elseBlock, ArrayRef<Value>());
251 
252  rewriter.eraseOp(ifOp);
253  return success();
254 }

260 static Optional<int32_t> getLaunchConfigIndex(Operation *op) {
261  auto dimAttr = op->getAttrOfType<StringAttr>("dimension");
262  if (!dimAttr) {
263  return {};
264  }
265  if (dimAttr.getValue() == "x") {
266  return 0;
267  } else if (dimAttr.getValue() == "y") {
268  return 1;
269  } else if (dimAttr.getValue() == "z") {
270  return 2;
271  }
272  return {};
273 }

275 template <typename SourceOp, spirv::BuiltIn builtin>
276 LogicalResult LaunchConfigConversion<SourceOp, builtin>::matchAndRewrite(
277  SourceOp op, ArrayRef<Value> operands,
278  ConversionPatternRewriter &rewriter) const {
279  auto index = getLaunchConfigIndex(op);
280  if (!index)
281  return failure();
282 
283  // SPIR-V invocation builtin variables are a vector of type <3xi32>
284  auto spirvBuiltin = spirv::getBuiltinVariableValue(op, builtin, rewriter);
285  rewriter.replaceOpWithNewOp<spirv::CompositeExtractOp>(
286  op, rewriter.getIntegerType(32), spirvBuiltin,
287  rewriter.getI32ArrayAttr({index.getValue()}));
288  return success();
289 }

306  return failure();
307 
308  auto workGroupSizeAttr = spirv::lookupLocalWorkGroupSize(op);

310  auto convertedType = typeConverter.convertType(op.getResult().getType());

322 // Legalizes a GPU function as an entry SPIR-V function.
323 static spirv::FuncOp
324 lowerAsEntryFunction(gpu::GPUFuncOp funcOp, SPIRVTypeConverter &typeConverter,
325  ConversionPatternRewriter &rewriter,
326  spirv::EntryPointABIAttr entryPointInfo,
327  ArrayRef<spirv::InterfaceVarABIAttr> argABIInfo) {
328  auto fnType = funcOp.getType();
329  if (fnType.getNumResults()) {
330  funcOp.emitError("SPIR-V lowering only supports entry functions"
331  "with no return values right now");
332  return nullptr;
333  }
334  if (fnType.getNumInputs() != argABIInfo.size()) {
335  funcOp.emitError(
336  "lowering as entry functions requires ABI info for all arguments");
337  return nullptr;
338  }
339  // Update the signature to valid SPIR-V types and add the ABI
340  // attributes. These will be "materialized" by using the
341  // LowerABIAttributesPass.
342  TypeConverter::SignatureConversion signatureConverter(fnType.getNumInputs());
343  {
344  for (auto argType : enumerate(funcOp.getType().getInputs())) {
345  auto convertedType = typeConverter.convertType(argType.value());
346  signatureConverter.addInputs(argType.index(), convertedType);
347  }
348  }
349  auto newFuncOp = rewriter.create<spirv::FuncOp>(
350  funcOp.getLoc(), funcOp.getName(),
351  rewriter.getFunctionType(signatureConverter.getConvertedTypes(),
352  llvm::None));
353  for (const auto &namedAttr : funcOp.getAttrs()) {
354  if (namedAttr.first == impl::getTypeAttrName() ||
355  namedAttr.first == SymbolTable::getSymbolAttrName())
356  continue;
357  newFuncOp.setAttr(namedAttr.first, namedAttr.second);
358  }
359  rewriter.inlineRegionBefore(funcOp.getBody(), newFuncOp.getBody(),
360  newFuncOp.end());
361  rewriter.applySignatureConversion(&newFuncOp.getBody(), signatureConverter);
362  rewriter.eraseOp(funcOp);
363 
364  spirv::setABIAttrs(newFuncOp, entryPointInfo, argABIInfo);
365  return newFuncOp;
366 }

368 /// Populates `argABI` with spv.interface_var_abi attributes for lowering
369 /// gpu.func to spv.func if no arguments have the attributes set
370 /// already. Returns failure if any argument has the ABI attribute set already.
371 static LogicalResult
372 getDefaultABIAttrs(MLIRContext *context, gpu::GPUFuncOp funcOp,
373  SmallVectorImpl<spirv::InterfaceVarABIAttr> &argABI) {
374  for (auto argIndex : llvm::seq<unsigned>(0, funcOp.getNumArguments())) {
375  if (funcOp.getArgAttrOfType<spirv::InterfaceVarABIAttr>(
376  argIndex, spirv::getInterfaceVarABIAttrName()))
377  return failure();
378  // Vulkan's interface variable requirements needs scalars to be wrapped in a
379  // struct. The struct held in storage buffer.
380  Optional<spirv::StorageClass> sc;
381  if (funcOp.getArgument(argIndex).getType().isIntOrIndexOrFloat())
382  sc = spirv::StorageClass::StorageBuffer;
383  argABI.push_back(spirv::getInterfaceVarABIAttr(0, argIndex, sc, context));
384  }
385  return success();
386 }

388 LogicalResult GPUFuncOpConversion::matchAndRewrite(
389  gpu::GPUFuncOp funcOp, ArrayRef<Value> operands,
390  ConversionPatternRewriter &rewriter) const {
391  if (!gpu::GPUDialect::isKernel(funcOp))
392  return failure();
393 
394  SmallVector<spirv::InterfaceVarABIAttr, 4> argABI;
395  if (failed(getDefaultABIAttrs(rewriter.getContext(), funcOp, argABI))) {
396  argABI.clear();
397  for (auto argIndex : llvm::seq<unsigned>(0, funcOp.getNumArguments())) {
398  // If the ABI is already specified, use it.
399  auto abiAttr = funcOp.getArgAttrOfType<spirv::InterfaceVarABIAttr>(
400  argIndex, spirv::getInterfaceVarABIAttrName());
401  if (!abiAttr) {
402  funcOp.emitRemark(
403  "match failure: missing 'spv.interface_var_abi' attribute at "
404  "argument ")
405  << argIndex;
406  return failure();
407  }
408  argABI.push_back(abiAttr);
409  }
410  }
411 
412  auto entryPointAttr = spirv::lookupEntryPointABI(funcOp);
413  if (!entryPointAttr) {
414  funcOp.emitRemark("match failure: missing 'spv.entry_point_abi' attribute");
415  return failure();
416  }
417  spirv::FuncOp newFuncOp = lowerAsEntryFunction(
418  funcOp, typeConverter, rewriter, entryPointAttr, argABI);
419  if (!newFuncOp)
420  return failure();
421  newFuncOp.removeAttr(Identifier::get(gpu::GPUDialect::getKernelFuncAttrName(),
422  rewriter.getContext()));
423  return success();
424 }

430 LogicalResult GPUModuleConversion::matchAndRewrite(
431  gpu::GPUModuleOp moduleOp, ArrayRef<Value> operands,
432  ConversionPatternRewriter &rewriter) const {
433  auto spvModule = rewriter.create<spirv::ModuleOp>(
434  moduleOp.getLoc(), spirv::AddressingModel::Logical,
435  spirv::MemoryModel::GLSL450);
436 
437  // Move the region from the module op into the SPIR-V module.
438  Region &spvModuleRegion = spvModule.body();
439  rewriter.inlineRegionBefore(moduleOp.body(), spvModuleRegion,
440  spvModuleRegion.begin());
441  // The spv.module build method adds a block with a terminator. Remove that
442  // block. The terminator of the module op in the remaining block will be
443  // legalized later.
444  rewriter.eraseBlock(&spvModuleRegion.back());
445  rewriter.eraseOp(moduleOp);
446  return success();
447 }
