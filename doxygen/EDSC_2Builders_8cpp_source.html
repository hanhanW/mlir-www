61 BlockHandle mlir::edsc::BlockHandle::create(ArrayRef<Type> argTypes) {
62  auto &currentB = ScopedContext::getBuilderRef();
63  auto *ib = currentB.getInsertionBlock();
64  auto ip = currentB.getInsertionPoint();
65  BlockHandle res;
66  res.block = ScopedContext::getBuilderRef().createBlock(ib->getParent());
67  // createBlock sets the insertion point inside the block.
68  // We do not want this behavior when using declarative builders with nesting.
69  currentB.setInsertionPoint(ib, ip);
70  for (auto t : argTypes) {
71  res.block->addArgument(t);
72  }
73  return res;
74 }

90 void mlir::edsc::LoopBuilder::operator()(function_ref<void(void)> fun) {
91  // Call to `exit` must be explicit and asymmetric (cannot happen in the
92  // destructor) because of ordering wrt comma operator.
93  /// The particular use case concerns nested blocks:
94  ///
95  /// ```c++
96  /// For (&i, lb, ub, 1)({
97  /// /--- destructor for this `For` is not always called before ...
98  /// V
99  /// For (&j1, lb, ub, 1)({
100  /// some_op_1,
101  /// }),
102  /// /--- ... this scope is entered, resulting in improperly nested IR.
103  /// V
104  /// For (&j2, lb, ub, 1)({
105  /// some_op_2,
106  /// }),
107  /// });
108  /// ```
109  if (fun)
110  fun();
111  exit();
112 }

119 mlir::edsc::BlockBuilder::BlockBuilder(BlockHandle *bh, ArrayRef<Type> types,
120  MutableArrayRef<Value> args) {
121  assert(!*bh && "BlockHandle already captures a block, use "
122  "the explicit BockBuilder(bh, Append())({}) syntax instead.");
123  assert((args.empty() || args.size() == types.size()) &&
124  "if args captures are specified, their number must match the number "
125  "of types");
126  *bh = BlockHandle::create(types);
127  if (!args.empty())
128  for (auto it : llvm::zip(args, bh->getBlock()->getArguments()))
129  std::get<0>(it) = Value(std::get<1>(it));
130  enter(bh->getBlock());
131 }
